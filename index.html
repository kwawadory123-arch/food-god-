// GameManager.cs
using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class GameManager : MonoBehaviour
{
    public static GameManager Instance { get; private set; }

    [Header("Economy")]
    public long currency = 50000;
    public long umami = 0;
    public long tapGain = 1;
    public long perSecondGain = 5;
    public float synergyMultiplier = 1f;

    [Header("References")]
    public CurrencyUIController currencyUI;
    public SaveManager saveManager;
    public AudioManager audioManager;
    public LoginBonusController loginBonusController;
    public GachaController gachaController;

    public event Action<long> OnCurrencyChanged;
    public event Action<long> OnUmamiChanged;
    public event Action<float> OnProductionChanged;

    private void Awake()
    {
        if (Instance != null && Instance != this) { Destroy(gameObject); return; }
        Instance = this;
        DontDestroyOnLoad(gameObject);
    }

    private void Start()
    {
        LoadGame();
        StartCoroutine(AutoProduceRoutine());
        StartCoroutine(AutoConvertUmamiRoutine());
        loginBonusController.TryShowLoginBonus();
    }

    private IEnumerator AutoProduceRoutine()
    {
        var wait = new WaitForSeconds(1f);
        while (true)
        {
            yield return wait;
            long gain = Mathf.CeilToInt((float)(perSecondGain * synergyMultiplier));
            AddCurrency(gain);
        }
    }

    private IEnumerator AutoConvertUmamiRoutine()
    {
        var wait = new WaitForSeconds(0.2f);
        while (true)
        {
            yield return wait;
            if (umami >= 50)
            {
                long bundles = umami / 50;
                long converted = bundles * 1000;
                umami -= bundles * 50;
                AddCurrency(converted);
                currencyUI.PlayBigGain(converted);
            }
        }
    }

    public void TapMainArea()
    {
        long gain = tapGain;
        AddCurrency(gain);
        currencyUI.SpawnTapText(gain);
        audioManager.PlayTap();
    }

    public void AddUmami(long amount)
    {
        umami += amount;
        OnUmamiChanged?.Invoke(umami);
        currencyUI.UpdateUmami(umami);
    }

    public bool CanSpend(long amount) => currency >= amount;

    public bool Spend(long amount)
    {
        if (currency < amount) return false;
        currency -= amount;
        OnCurrencyChanged?.Invoke(currency);
        currencyUI.UpdateCurrency(currency);
        return true;
    }

    public void AddCurrency(long amount)
    {
        currency += amount;
        OnCurrencyChanged?.Invoke(currency);
        currencyUI.UpdateCurrency(currency);
    }

    public void SetProduction(long perSec)
    {
        perSecondGain = perSec;
        OnProductionChanged?.Invoke(perSecondGain);
        currencyUI.UpdateProduction(perSecondGain, tapGain, synergyMultiplier);
    }

    public void SetSynergyMultiplier(float mult)
    {
        synergyMultiplier = mult;
        OnProductionChanged?.Invoke(perSecondGain);
        currencyUI.UpdateProduction(perSecondGain, tapGain, synergyMultiplier);
    }

    public void SaveGame()
    {
        saveManager.Save(currency, umami, BoardAreaManager.Instance.ExportBoardState());
    }

    public void LoadGame()
    {
        var data = saveManager.Load();
        currency = data.currencyValue;
        umami = data.umamiValue;
        currencyUI.UpdateAll(currency, umami, perSecondGain, tapGain, synergyMultiplier);
    }

    private void OnApplicationPause(bool pause)
    {
        if (pause) SaveGame();
    }

    private void OnApplicationQuit()
    {
        SaveGame();
    }
}

using UnityEngine;

[RequireComponent(typeof(Collider2D))]
public class GodInstance : MonoBehaviour, IBeginDragHandler, IDragHandler, IEndDragHandler, IPointerDownHandler, IPointerUpHandler
{
    public GodData data;
    public int currentRank;
    public bool isDragging;
    public bool isInBoard;
    public Transform visualRoot;

    private Vector3 startPos;
    private CanvasGroup canvasGroup;
    private BoardAreaManager board;
    private DragHandler dragHandler;
    private float longPressTime;
    private bool longPressed;

    private void Awake()
    {
        canvasGroup = GetComponent<CanvasGroup>();
        if (!canvasGroup) canvasGroup = gameObject.AddComponent<CanvasGroup>();
        board = BoardAreaManager.Instance;
    }

    public void Init(GodData god)
    {
        data = god;
        currentRank = god.rank;
        RefreshVisual();
    }

    public void RefreshVisual()
    {
        if (visualRoot != null)
        {
            visualRoot.localScale = Vector3.one;
        }
    }

    public void OnPointerDown(PointerEventData eventData)
    {
        startPos = transform.position;
        longPressTime = Time.unscaledTime;
        longPressed = false;
        StartCoroutine(LongPressRoutine());
    }

    public void OnPointerUp(PointerEventData eventData)
    {
        if (!isDragging)
        {
            transform.localScale = Vector3.one;
        }
    }

    private IEnumerator LongPressRoutine()
    {
        yield return new WaitForSecondsRealtime(0.55f);
        if (!isDragging && Time.unscaledTime - longPressTime >= 0.55f)
        {
            longPressed = true;
            PopupController.Instance.ShowGodDetail(this);
            transform.position += Vector3.up * 0.12f;
        }
    }

    public void OnBeginDrag(PointerEventData eventData)
    {
        isDragging = true;
        canvasGroup.alpha = 0.6f;
        transform.localScale = Vector3.one * 1.08f;
        board.BeginDrag(this);
    }

    public void OnDrag(PointerEventData eventData)
    {
        Vector3 world = Camera.main.ScreenToWorldPoint(eventData.position);
        transform.position = new Vector3(world.x, world.y, 0f);
        board.UpdateDragHighlight(this);
    }

    public void OnEndDrag(PointerEventData eventData)
    {
        isDragging = false;
        canvasGroup.alpha = 1f;
        transform.localScale = Vector3.one;
        board.EndDrag(this);
    }

    public void ReturnToStart()
    {
        transform.position = startPos;
    }

    public void PlaySpawnBounce()
    {
        StartCoroutine(BounceRoutine());
    }

    private IEnumerator BounceRoutine()
    {
        Vector3 baseScale = transform.localScale;
        transform.localScale = baseScale * 0.9f;
        yield return new WaitForSeconds(0.08f);
        transform.localScale = baseScale * 1.08f;
        yield return new WaitForSeconds(0.08f);
        transform.localScale = baseScale;
    }
}

// BoardAreaManager.cs
using System.Collections.Generic;
using UnityEngine;

public class BoardAreaManager : MonoBehaviour
{
    public static BoardAreaManager Instance { get; private set; }

    [System.Serializable]
    public class BoardSlot
    {
        public Transform point;
        public GodInstance current;
    }

    public List<BoardSlot> slots = new List<BoardSlot>();
    public float mergeDistance = 0.85f;
    public MergeSystem mergeSystem;
    public SynergySystem synergySystem;
    public RareEffectController rareEffectController;

    private void Awake()
    {
        Instance = this;
    }

    public bool TryPlaceGod(GodInstance god)
    {
        BoardSlot nearest = FindNearestEmptySlot(god.transform.position);
        if (nearest == null) return false;
        nearest.current = god;
        god.isInBoard = true;
        god.transform.position = nearest.point.position;
        god.PlaySpawnBounce();
        rareEffectController.PlayPlacementEffect(god);
        return true;
    }

    public BoardSlot FindNearestEmptySlot(Vector3 pos)
    {
        BoardSlot best = null;
        float bestDist = float.MaxValue;
        foreach (var slot in slots)
        {
            if (slot.current != null) continue;
            float d = Vector3.Distance(pos, slot.point.position);
            if (d < bestDist)
            {
                bestDist = d;
                best = slot;
            }
        }
        return best;
    }

    public void BeginDrag(GodInstance god)
    {
        synergySystem.SetPreview(god, true);
    }

    public void UpdateDragHighlight(GodInstance god)
    {
        var target = FindMergeTarget(god);
        mergeSystem.SetHighlightTarget(target);
    }

    public void EndDrag(GodInstance god)
    {
        synergySystem.SetPreview(god, false);
        var target = FindMergeTarget(god);
        if (target != null)
        {
            mergeSystem.Merge(god, target);
            return;
        }
        if (!TryPlaceGod(god))
            god.ReturnToStart();
        synergySystem.Recalculate();
    }

    public GodInstance FindMergeTarget(GodInstance source)
    {
        foreach (var slot in slots)
        {
            if (slot.current == null || slot.current == source) continue;
            if (slot.current.data.id == source.data.id && slot.current.currentRank == source.currentRank)
            {
                if (Vector3.Distance(slot.current.transform.position, source.transform.position) < mergeDistance)
                    return slot.current;
            }
        }
        return null;
    }

    public List<string> ExportBoardState()
    {
        var list = new List<string>();
        foreach (var slot in slots)
        {
            list.Add(slot.current != null ? slot.current.data.id : "");
        }
        return list;
    }
}

// MergeSystem.cs
using System.Collections;
using UnityEngine;

public class MergeSystem : MonoBehaviour
{
    public MergeResultPopup mergePopup;
    public RareEffectController rareEffectController;
    public AudioManager audioManager;
    public CameraShake cameraShake;

    private GodInstance highlighted;

    public void SetHighlightTarget(GodInstance target)
    {
        if (highlighted == target) return;
        if (highlighted != null) highlighted.transform.localScale = Vector3.one;
        highlighted = target;
        if (highlighted != null) highlighted.transform.localScale = Vector3.one * 1.08f;
    }

    public void Merge(GodInstance a, GodInstance b)
    {
        StartCoroutine(MergeRoutine(a, b));
    }

    private IEnumerator MergeRoutine(GodInstance a, GodInstance b)
    {
        audioManager.PlayMerge();
        rareEffectController.PlayMergeBurst(a.transform.position);
        cameraShake.Shake(0.18f, 0.12f);

        Vector3 center = (a.transform.position + b.transform.position) * 0.5f;
        yield return new WaitForSeconds(0.1f);

        Destroy(b.gameObject);
        var nextData = GodDatabase.Instance.GetNextEvolution(a.data);
        if (nextData == null)
        {
            yield break;
        }

        a.data = nextData;
        a.currentRank += 1;
        a.transform.position = center;
        a.PlaySpawnBounce();

        mergePopup.Show(nextData.displayName, nextData.icon, nextData.rarity);
        SynergySystem.Instance.Recalculate();
    }
}

// SynergySystem.cs
using System.Collections.Generic;
using System.Linq;
using UnityEngine;

public class SynergySystem : MonoBehaviour
{
    public static SynergySystem Instance { get; private set; }

    public List<SynergyData> synergyDatas = new List<SynergyData>();
    public LineRenderer linePrefab;
    public Transform effectRoot;
    public CurrencyUIController currencyUI;

    private readonly List<LineRenderer> activeLines = new List<LineRenderer>();

    private void Awake()
    {
        Instance = this;
    }

    public void SetPreview(GodInstance god, bool active)
    {
        if (!active) ClearPreview();
    }

    public void Recalculate()
    {
        ClearLines();
        float multiplier = 1f;
        var gods = BoardAreaManager.Instance.slots.Where(s => s.current != null).Select(s => s.current).ToList();

        foreach (var synergy in synergyDatas)
        {
            var matched = gods.Where(g => synergy.requiredTags.All(t => g.data.synergyTags.Contains(t))).ToList();
            if (matched.Count >= 2)
            {
                multiplier *= synergy.bonusMultiplier;
                DrawSynergyLine(matched[0], matched[1], synergy);
            }
        }

        GameManager.Instance.SetSynergyMultiplier(multiplier);
    }

    private void DrawSynergyLine(GodInstance a, GodInstance b, SynergyData synergy)
    {
        var line = Instantiate(linePrefab, effectRoot);
        line.positionCount = 2;
        line.useWorldSpace = true;
        line.SetPosition(0, a.transform.position);
        line.SetPosition(1, b.transform.position);
        activeLines.Add(line);
    }

    private void ClearLines()
    {
        foreach (var l in activeLines)
        {
            if (l != null) Destroy(l.gameObject);
        }
        activeLines.Clear();
    }

    private void ClearPreview()
    {
    }
}

        {

            umami = umami.ToString(),
            lastSaveUnix = DateTimeOffset.Now.ToUnixTimeSeconds(),
            boardGodIds = board
        };
        PlayerPrefs.SetString(Key, JsonUtility.ToJson(data));
        PlayerPrefs.Save();
    }

    public LoadedGameData Load()
    {
        if (!PlayerPrefs.HasKey(Key))
        {
            return new LoadedGameData { currencyValue = 50000, umamiValue = 0 };
        }

        var json = PlayerPrefs.GetString(Key);
        var data = JsonUtility.FromJson<SaveData>(json);
        return new LoadedGameData
        {
            currencyValue = long.TryParse(data.currency, out var c) ? c : 50000,
            umamiValue = long.TryParse(data.umami, out var u) ? u : 0
        };
    }
}

// NumberFormatUtil.cs
using System.Globalization;

public static class NumberFormatUtil
{
    public static string Format(long value)
    {
        if (value < 1000) return value.ToString();
        string[] suffix = { "", "K", "M", "B", "T", "Q" };
        double v = value;
        int i = 0;
        while (v >= 1000 && i < suffix.Length - 1)
        {
            v /= 1000;
            i++;
        }
        return v.ToString(v >= 100 ? "0" : v >= 10 ? "0.0" : "0.00", CultureInfo.InvariantCulture) + suffix[i];
    }
}

// OfflineRewardCalculator.cs
using System;

public static class OfflineRewardCalculator
{
    public static long CalculateReward(DateTime last, DateTime now, long perSecond)
    {
        var seconds = Math.Max(0, (now - last).TotalSeconds);
        seconds = Math.Min(seconds, 60 * 60 * 8);
        return (long)(seconds * perSecond * 0.3f);
    }
}

5. ガチャ設計

朝は normalgacha で ¥1000、昼は spgacha で ¥100000、特定時間帯は SecretZ Gacha を出し、所持金が KB 以上なら表示します。

// GachaController.cs
using System;
using System.Collections.Generic;
using UnityEngine;

public class GachaController : MonoBehaviour
{
    public Sprite normalGachaCover;
    public Sprite specialGachaCover;
    public Sprite secretZGachaCover;

    public long normalPrice = 1000;
    public long specialPrice = 100000;

    public List<GodData> normalPool = new List<GodData>();
    public List<GodData> specialPool = new List<GodData>();
    public List<GodData> secretZPool = new List<GodData>();

    public string GetCurrentGachaId()
    {
        var hour = DateTime.Now.Hour;
        if ((hour >= 6 && hour < 8) || (hour >= 17 && hour < 20))
            return "secretz";
        return hour < 12 ? "normalgacha" : "spgacha";
    }

    public long GetCurrentPrice()
    {
        string id = GetCurrentGachaId();
        if (id == "normalgacha") return normalPrice;
        if (id == "spgacha") return specialPrice;
        return 0;
    }

    public Sprite GetCurrentCover()
    {
        string id = GetCurrentGachaId();
        if (id == "normalgacha") return normalGachaCover;
        if (id == "spgacha") return specialGachaCover;
        return secretZGachaCover;
    }

    public GodData Roll()
    {
        string id = GetCurrentGachaId();
        if (id == "normalgacha") return normalPool[UnityEngine.Random.Range(0, normalPool.Count)];
        if (id == "spgacha") return specialPool[UnityEngine.Random.Range(0, specialPool.Count)];
        return secretZPool[UnityEngine.Random.Range(0, secretZPool.Count)];
    }
}

6. UI構成手順

  1. Canvas を作成し、MainGameCanvas と PopupCanvas を分ける。
  2. メインUIに CurrencyText、UmamiText、PerSecondText、TapGainText、SynergyText を配置する。
  3. 画面中央にタップ受けエリアを置き、TapMainArea を呼ぶ。
  4. 神域は Grid/Anchor 付きの大きめの BoardRoot を用意する。
  5. 各ポップアップは CanvasGroup 付きで、開閉をアニメーション化する。

7. Inspector設定例

  • GameManager に CurrencyUIController、SaveManager、AudioManager、LoginBonusController を接続。
  • BoardAreaManager に slot 位置を 9〜16 個ほど登録。
  • MergeSystem に MergeResultPopup、RareEffectController、CameraShake を接続。
  • SynergySystem に SynergyData を 3〜5 個以上登録。
  • GachaController に normal / special / secretZ の各プールを登録。

8. 仮素材向け演出

仮素材でも成立するよう、各 GodData に emojiFallback を設定してください。アイコン未設定時は絵文字を表示し、レア度に応じて色と発光量だけ変える構成です。

9. 拡張ポイント

  • GodData を ScriptableObject アセット化してキャラ追加を容易にする。
  • シナジーをタグベースで増やす。
  • ログボや日課を毎日更新に拡張する。
  • 限定キャラのガチャ排出率を時間帯ごとに変える。
  • MergeResultPopup を Timeline / Animator 化する。
